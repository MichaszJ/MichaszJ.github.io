# glider material densities
density_foam = 25.5
density_16balsa = 0.00030162499999999997
density_8balsa = 0.0018097499999999997
density_cf = 1377.964875
density_abs = 1040

# masses of individual components
glider_masses = [
    density_foam * 1.192e-4, # fuselage
    density_16balsa * 5.549e-6, # winglets
    density_cf * 5.645e-6, # tail boom
    density_8balsa * 6.900e-6, # vertical tail support
    density_16balsa * 8.593e-6, # top tail
    density_abs * 3.483e-6 # tail insert
]

# volume of composite 
skin_outside = 0.0003048 * 0.145
wing_outer = skin_outside*1440

# inner section of wing, density = 32.0369 kg/m^3
wing_inner = 32.0369 * 1.311e-4

glider_mass = sum(glider_masses) # includes wing inside
